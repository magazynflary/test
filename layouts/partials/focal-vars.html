{{- $focus := .focus | default "50% 50%" -}} {{- $focusParts := split $focus " " -}} {{- $objectPosition := printf "%s %s" (index $focusParts 0) (index $focusParts 1) -}} {{- $zoom := cond (gt (len $focusParts) 2) (index $focusParts 2) "" -}} {{- $vars := printf "--focal-pos: %s" $objectPosition -}} {{- if $zoom -}}{{- $vars = printf "%s; --focal-zoom: %s" $vars $zoom -}}{{- end -}} {{- return $vars -}}
